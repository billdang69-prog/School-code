import random
# initialize the game
def math_game():
    print("Welcome to the Math Game!")
    print("You will get random addition, subtraction, multiplication, or division questions.")
    print("Type 'exit' anytime to quit.\n")

    score = 0

    while True:
        num1 = random.randint(1, 10)
        num2 = random.randint(1, 10)

        # Randomly choose an operation
        operation = random.choice(["+", "-", "×", "÷"])

        # Make division clean
        if operation == "÷":
            num1 = num1 * num2
            correct = num1 // num2
        elif operation == "+":
            correct = num1 + num2
        elif operation == "-":
            correct = num1 - num2
        elif operation == "×":
            correct = num1 * num2

        # Ask the question
        answer = input(f"What is {num1} {operation} {num2}? ")

        if answer.lower() == "exit":
            break

        try:
            if int(answer) == correct:
                print("Correct!\n")
                score += 1
            else:
                print(f"Incorrect. The correct answer was {correct}.\n")
        except ValueError:
            print("Please enter a number.\n")

    print(f"Game over! Your final score is {score}")

math_game()
